function loginUser(){ let u = document.getElementById("userName").value.trim() let p = document.getElementById("userPass").value.trim() let msg = document.getElementById("loginMsg") if(!u || !p){ msg.innerText = "Enter username & password" return } let db = JSON.parse(localStorage.getItem("alphabet_users") || "{}") // register if(!db[u]){ db[u] = { password: p, bestScore: 0 } localStorage.setItem("alphabet_users", JSON.stringify(db)) } // login check if(db[u].password !== p){ msg.innerText = "Wrong password" return } CURRENT_USER = u document.getElementById("loginScreen").style.display = "none" renderLeaderboard() }
